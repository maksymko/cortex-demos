124class RegSetClearStub : public IOHandlerStub {
125 public:
133 RegSetClearStub(uint32_t addr, uint32_t set_addr, uint32_t clr_addr) :
134 rw_addr_{addr}, set_addr_{set_addr}, clr_addr_{clr_addr} {}
135
136 uint32_t write32(uint32_t addr, uint32_t old_value, uint32_t new_value) override {
137 auto write_value = new_value;
138 if (addr == set_addr_) {
139 write_value = (old_value | new_value);
140 } else if (addr == clr_addr_) {
141 write_value = (old_value & (~new_value));
142 }
143
144 if (addr != rw_addr_) {
145 set_mem_value(rw_addr_, write_value);
146 set_mem_value(set_addr_, write_value);
147 set_mem_value(clr_addr_, write_value);
148 }
149 return write_value;
150 }
151
152 uint32_t read32(uint32_t addr, uint32_t value) override {
153 if (addr == set_addr_ || addr == clr_addr_) {
154 return get_mem_value(rw_addr_);
155 }
156
157 return value;
158 }
159
160 protected:
161 const uint32_t rw_addr_;
162 const uint32_t set_addr_;
163 const uint32_t clr_addr_;
164};
